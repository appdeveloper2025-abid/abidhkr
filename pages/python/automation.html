File Management

import os import shutil from pathlib import Path def organize_files_by_extension(directory): """Organize files in directory by extension""" path = Path(directory) for file in path.iterdir(): if file.is_file(): extension = file.suffix.lower() if extension: # Create directory for extension ext_dir = path / extension[1:] # Remove the dot ext_dir.mkdir(exist_ok=True) # Move file to extension directory shutil.move(str(file), str(ext_dir / file.name)) def backup_important_files(source_dir, backup_dir): """Backup files with specific extensions""" important_extensions = ['.py', '.txt', '.md', '.json', '.xml'] backup_path = Path(backup_dir) backup_path.mkdir(exist_ok=True) for ext in important_extensions: files = Path(source_dir).glob(f'**/*{ext}') for file in files: dest = backup_path / file.name shutil.copy2(file, dest) print(f"Backed up: {file.name}") # Usage organize_files_by_extension("/path/to/downloads") backup_important_files("/home/user/projects", "/backup/projects")

Log Analysis

import re from collections import Counter def analyze_access_logs(log_file): """Analyze web server access logs""" ip_pattern = r'^(\d+\.\d+\.\d+\.\d+)' status_pattern = r'" (\d{3}) ' ips = [] status_codes = [] with open(log_file, 'r') as f: for line in f: # Extract IP addresses ip_match = re.search(ip_pattern, line) if ip_match: ips.append(ip_match.group(1)) # Extract status codes status_match = re.search(status_pattern, line) if status_match: status_codes.append(status_match.group(1)) # Count occurrences ip_counts = Counter(ips) status_counts = Counter(status_codes) print("Top 10 IP addresses:") for ip, count in ip_counts.most_common(10): print(f"{ip}: {count}") print("\nStatus code distribution:") for status, count in status_counts.items(): print(f"{status}: {count}") # Usage analyze_access_logs("/var/log/apache2/access.log")

Port Scanner

import socket import threading from datetime import datetime class PortScanner: def __init__(self, target, threads=100): self.target = target self.threads = threads self.open_ports = [] self.lock = threading.Lock() def scan_port(self, port): try: sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM) sock.settimeout(1) result = sock.connect_ex((self.target, port)) if result == 0: with self.lock: self.open_ports.append(port) print(f"Port {port}: Open") sock.close() except: pass def scan_range(self, start_port, end_port): print(f"Scanning {self.target} from port {start_port} to {end_port}") start_time = datetime.now() threads = [] for port in range(start_port, end_port + 1): thread = threading.Thread(target=self.scan_port, args=(port,)) threads.append(thread) thread.start() # Limit concurrent threads if len(threads) >= self.threads: for t in threads: t.join() threads = [] # Wait for remaining threads for t in threads: t.join() end_time = datetime.now() print(f"Scan completed in {end_time - start_time}") return self.open_ports # Usage scanner = PortScanner("192.168.1.1") open_ports = scanner.scan_range(1, 1000)

Website Monitor

import requests import time import smtplib from email.mime.text import MIMEText class WebsiteMonitor: def __init__(self, urls, check_interval=300): self.urls = urls self.check_interval = check_interval self.status_history = {} def check_website(self, url): try: response = requests.get(url, timeout=10) status = { 'url': url, 'status_code': response.status_code, 'response_time': response.elapsed.total_seconds(), 'timestamp': time.time(), 'available': response.status_code == 200 } return status except Exception as e: return { 'url': url, 'status_code': None, 'response_time': None, 'timestamp': time.time(), 'available': False, 'error': str(e) } def send_alert(self, message): # Configure email settings smtp_server = "smtp.gmail.com" smtp_port = 587 email = "your_email@gmail.com" password = "your_password" msg = MIMEText(message) msg['Subject'] = "Website Alert" msg['From'] = email msg['To'] = "admin@company.com" try: server = smtplib.SMTP(smtp_server, smtp_port) server.starttls() server.login(email, password) server.send_message(msg) server.quit() except Exception as e: print(f"Failed to send email: {e}") def monitor(self): while True: for url in self.urls: status = self.check_website(url) if not status['available']: message = f"Website {url} is down! Error: {status.get('error', 'Unknown')}" print(message) self.send_alert(message) else: print(f"{url} is up (Response time: {status['response_time']:.2f}s)") time.sleep(self.check_interval) # Usage urls = ["https://example.com", "https://google.com"] monitor = WebsiteMonitor(urls, check_interval=60) # monitor.monitor() # Uncomment to start monitoring

Automated Backup Script

import os import shutil import schedule import time from datetime import datetime def create_backup(): """Create automated backup of important directories""" source_dirs = [ "/home/user/Documents", "/home/user/Scripts", "/etc" ] backup_base = "/backup" timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") for source in source_dirs: if os.path.exists(source): dir_name = os.path.basename(source) backup_dir = f"{backup_base}/{dir_name}_{timestamp}" try: shutil.copytree(source, backup_dir) print(f"Backup created: {backup_dir}") except Exception as e: print(f"Backup failed for {source}: {e}") # Schedule backups schedule.every().day.at("02:00").do(create_backup) schedule.every().sunday.at("01:00").do(create_backup) # Run scheduler while True: schedule.run_pending() time.sleep(60)

System Health Monitor

import psutil import time import json from datetime import datetime class SystemMonitor: def __init__(self, log_file="system_health.json"): self.log_file = log_file self.alerts = [] def get_system_stats(self): """Collect system statistics""" return { 'timestamp': datetime.now().isoformat(), 'cpu_percent': psutil.cpu_percent(interval=1), 'memory_percent': psutil.virtual_memory().percent, 'disk_percent': psutil.disk_usage('/').percent, 'network_io': psutil.net_io_counters()._asdict(), 'process_count': len(psutil.pids()) } def check_thresholds(self, stats): """Check if any metrics exceed thresholds""" if stats['cpu_percent'] > 80: self.alerts.append(f"High CPU usage: {stats['cpu_percent']}%") if stats['memory_percent'] > 85: self.alerts.append(f"High memory usage: {stats['memory_percent']}%") if stats['disk_percent'] > 90: self.alerts.append(f"High disk usage: {stats['disk_percent']}%") def log_stats(self, stats): """Log statistics to file""" with open(self.log_file, 'a') as f: f.write(json.dumps(stats) + '\n') def monitor(self, interval=60): """Start monitoring system""" while True: stats = self.get_system_stats() self.check_thresholds(stats) self.log_stats(stats) if self.alerts: print(f"ALERTS: {', '.join(self.alerts)}") self.alerts.clear() time.sleep(interval) # Usage monitor = SystemMonitor() # monitor.monitor() # Uncomment to start monitoring
